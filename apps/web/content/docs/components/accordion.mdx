## Usage

<hr className="mt-6 dark:border-neutral-800" />

```tsx
import {
  Accordion,
  AccordionItem,
  AccordionTrigger,
  AccordionContent,
} from "@sihaxito/deluxe-ui";
```

```tsx showLineNumbers
<Accordion type="single">
  <AccordionItem>
    <AccordionTrigger value="element-1">Element 1</AccordionTrigger>
    <AccordionContent>
      Lorem ipsum dolor sit amet, consectetur adipiscing elit.
    </AccordionContent>
  </AccordionItem>
</Accordion>
```
